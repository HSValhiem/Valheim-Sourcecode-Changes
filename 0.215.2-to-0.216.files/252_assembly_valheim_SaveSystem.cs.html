      public static string NormalizePath(string path, FileHelpers.FileSource source)
      {
          char[] array = new char[path.Length];
          int num = 0;
          int i = 0;
          while (i < path.Length)
          {
              char c = path[i];
              if (c == '\\')
              {
                  c = '/';
              }
              if (c != '/')
              {
                  goto IL_3A;
              }
              if (num > 0)
              {
                  if (array[num - 1] != '/')
                  {
                      goto IL_3A;
                  }
              }
              else if (source != FileHelpers.FileSource.Cloud)
              {
                  goto IL_3A;
              }
              IL_42:
              i++;
              continue;
              IL_3A:
              array[num++] = c;
              goto IL_42;
          }
          return new string(array, 0, num);
      }
